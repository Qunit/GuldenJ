package org.guldenj.core.listeners;

import org.guldenj.core.Coin;
import org.guldenj.core.ECKey;
import org.guldenj.core.Transaction;
import org.guldenj.core.Wallet;
import org.guldenj.script.Script;
import org.guldenj.wallet.AbstractKeyChainEventListener;

import java.util.List;

/**
 * Deprecated: implement the more specific event listener interfaces instead.
 */
@Deprecated
public abstract class AbstractWalletEventListener extends AbstractKeyChainEventListener implements WalletEventListener {
    @Override
    public void onCoinsReceived(Wallet wallet, Transaction tx, Coin prevBalance, Coin newBalance) {
        onChange();
    }

    @Override
    public void onCoinsSent(Wallet wallet, Transaction tx, Coin prevBalance, Coin newBalance) {
        onChange();
    }

    @Override
    public void onReorganize(Wallet wallet) {
        onChange();
    }

    @Override
    public void onTransactionConfidenceChanged(Wallet wallet, Transaction tx) {
        onChange();
    }

    @Override
    public void onKeysAdded(List<ECKey> keys) {
        onChange();
    }

    @Override
    public void onScriptsChanged(Wallet wallet, List<Script> scripts, boolean isAddingScripts) {
        onChange();
    }

    @Override
    public void onWalletChanged(Wallet wallet) {
        onChange();
    }

    /**
     * Default method called on change events.
     */
    public void onChange() {
    }
}
